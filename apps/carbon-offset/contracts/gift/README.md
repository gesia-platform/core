// SENDER WALLET CODE
    const Web3 = require("web3");
    const web3 = new Web3(new Web3.providers.HttpProvider("RPC_URL"));
    const SENDER_PRIVATE_KEY = ""; // Sender wallet private key
    const CONTRACT_ADDRESS = ""; // NFT Voucher Contract address

async function signTransfer(from, to, nonce,tokenId, amount) {
    amount = web3.utils.padLeft(web3.utils.toHex(amount), 64);
    nonce = web3.utils.padLeft(web3.utils.toHex(nonce), 64);
    tokenId = web3.utils.padLeft(web3.utils.toHex(tokenId), 64);
    const combined = web3.utils.encodePacked(from, to, tokenId, amount, nonce, CONTRACT_ADDRESS);
    const hashMessage = web3.utils.keccak256(combined);
    const signature = await web3.eth.accounts.sign(hashMessage, SENDER_PRIVATE_KEY);
    console.log("Hash Message : " + hashMessage);
    console.log("from : " + from);
    console.log("to : " + to);
    console.log("amount : " + amount);
    console.log("nonce : " + nonce);
    console.log("contract_address : " + CONTRACT_ADDRESS);
    return signature;
}

// RECEIVER CODE
    const Web3 = require("web3");
    const web3 = new Web3(new Web3.providers.HttpProvider("RPC_URL"));
    const RECEIVER_PRIVATE_KEY = ""; // Receiver wallet private key
    const CONTRACT_ADDRESS = ""; // NFT Voucher Contract address
    const CONTRACT_ABI = [];

    const contract = new web3.eth.Contract(CONTRACT_ABI, CONTRACT_ADDRESS);


    const from = "0x68cb4D2Da9323586c11D58cc3C22F96282319050";
    const to = "0xA5E91E9CD16E7657Da91A30C29aA7C3ce2C3c1ac";
    const nonce = 1001;
    const tokenId = 1;
    const amount = Web3.utils.toWei("1", "ether");
    const signature = "0x0b6b0a707b127324bde587cc6b3ddcc35821cf8db90fbeb53610502ed326ea85";  // The signature generated by senderSign.js


    const data = contract.methods.transferWithSignature(from, to, tokenId, amount, nonce, signature).encodeABI();
    const tx = {
        to: CONTRACT_ADDRESS,
        gas: 2000000,
        data: data,
    };

    const signedTx = await web3.eth.accounts.signTransaction(tx, RECEIVER_PRIVATE_KEY);
    const receipt = await web3.eth.sendSignedTransaction(signedTx.rawTransaction);
    console.log(`Transaction hash: ${receipt.transactionHash}`);
